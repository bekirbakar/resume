% This LaTeX source code creates a document listing academic publications for a researcher.
% The document is formatted as an article and provides a clear and concise layout.

% Load necessary packages.
\documentclass[a4paper,12pt]{article} % Set document class, paper size, and font size.
\usepackage[T1]{fontenc} % Set font encoding.
\usepackage[utf8]{inputenc} % Set input encoding.
\usepackage{graphicx} % Package for handling images.
\usepackage[hidelinks]{hyperref} % Package for hyperlinks with hidden link borders.
\usepackage[margin=1in]{geometry} % Package for setting page margins.
\usepackage{fancyhdr} % Package for customizing headers and footers.
\usepackage{lastpage} % Package for getting the number of the last page.
\usepackage{enumitem} % Package for customizing lists.
\usepackage{xcolor} % Package for handling colors.
\usepackage{titlesec} % Package for customizing section spacing.
\usepackage[sorting=ydnt,backend=biber,maxbibnames=99]{biblatex} % Package for managing bibliography.
\BiblatexSplitbibDefernumbersWarningOff
\usepackage{times} % Package for using Times New Roman font.
\usepackage{url} % Package for breaking long URLs.
\usepackage{tikz} % Package for drawing figures.

\addbibresource{../assets/bibliography.bib} % Add the bibliography file.

\titleformat{\section}{\vspace{1em}\bfseries\scshape\Large}{}{0em}{}
\titlespacing*{\section}{0pt}{1em}{0.5em}

\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}

\hypersetup{colorlinks=true,linkcolor=blue,urlcolor=blue}

\definecolor{sectioncolor}{RGB}{0, 0, 0}

\titleformat{\section}{\color{sectioncolor}\Large\bfseries}{\thesection}{1em}{}[\vspace{-12pt}\color{sectioncolor}\rule{\linewidth}{0.1pt}\vspace{-6pt}]

    \end{tikzpicture}
